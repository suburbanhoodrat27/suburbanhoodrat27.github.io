        function togglePolaroid() { document.getElementById('polaroid-container').classList.toggle('polaroid-active'); }


        const nav = document.getElementById('station-nav');

        const vLine = document.getElementById('v-line');

        const hLine = document.getElementById('h-line');

        const mainEl = document.querySelector('main');

        const headerEl = document.getElementById('page-header');

        const compactBrand = document.getElementById('compact-brand');


        function syncLines() {

            if (nav.classList.contains('retracted')) return;

            const navRect = nav.getBoundingClientRect();

            vLine.style.left = (navRect.left + 32) + 'px';

            const activeDisc = document.querySelector('.station-link.scrolled-active .disc') || nav.querySelector('.disc');

            if (activeDisc) hLine.style.top = (activeDisc.getBoundingClientRect().top + window.scrollY + (activeDisc.offsetHeight / 2)) + 'px';

        }


        function dockLine(element) {

            if (nav.classList.contains('retracted')) return;

            const disc = element.querySelector('.disc');

            if(!disc) return;

            const rect = disc.getBoundingClientRect();

            hLine.style.top = (rect.top + window.scrollY + (rect.height / 2)) + 'px';

            const discColor = window.getComputedStyle(disc).backgroundColor;

            hLine.style.backgroundColor = discColor;

        }


        function goHome() {

            nav.classList.remove('retracted');

            headerEl.classList.remove('opacity-0', 'pointer-events-none');

            compactBrand.classList.add('opacity-0', 'pointer-events-none');

            showView('home-view');

            closeVideo();

            setTimeout(syncLines, 100);

        }


        function navigateTo(view, colorClass, element) {

            // Instant retraction state to prevent jumping

            nav.classList.add('retracted');

            headerEl.classList.add('opacity-0', 'pointer-events-none');

            compactBrand.classList.remove('opacity-0', 'pointer-events-none');


            mainEl.classList.add('shunting');

            setTimeout(() => {

                if (view === 'Work') showView('work-view');

                else if (view === 'About') showView('about-view');

                else {

                    document.getElementById('generic-title').innerText = view;

                    showView('generic-view');

                }

                mainEl.classList.remove('shunting');

            }, 500);

        }


        function showView(id) {

            ['home-view', 'work-view', 'about-view', 'generic-view'].forEach(v => document.getElementById(v).classList.add('hidden'));

            document.getElementById(id).classList.remove('hidden');

        }


        function playVideo(title, youtubeId) {

            const iframe = document.getElementById('youtube-iframe');

            const placeholder = document.getElementById('video-placeholder');

            

            if (youtubeId) {

                // Using exact embed structure provided by user for maximum compatibility

                iframe.src = `https://www.youtube.com/embed/${youtubeId}?autoplay=1&si=GwwyW3wUOnB-8y0X`;

                iframe.classList.remove('hidden');

                placeholder.classList.add('hidden');

            } else {

                iframe.src = "";

                iframe.classList.add('hidden');

                placeholder.classList.remove('hidden');

                document.getElementById('current-vid-title').innerText = title;

            }

            

            document.getElementById('video-grid').classList.add('hidden');

            document.getElementById('full-player').classList.remove('hidden');

            document.getElementById('work-view').classList.add('video-active');

        }


        function closeVideo() {

            const iframe = document.getElementById('youtube-iframe');

            iframe.src = "";

            document.getElementById('video-grid').classList.remove('hidden');

            document.getElementById('full-player').classList.add('hidden');

            document.getElementById('work-view').classList.remove('video-active');

        }

            syncLines();

            window.addEventListener('resize', syncLines);

            window.addEventListener('scroll', syncLines);

            const observer = new IntersectionObserver((entries) => {

                entries.forEach(entry => {

                    if (entry.isIntersecting) {

                        entry.target.classList.add('scrolled-active');

                        dockLine(entry.target);

                    } else {

                        entry.target.classList.remove('scrolled-active');

                    }

                });

            }, { root: null, rootMargin: '-40% 0% -40% 0%', threshold: 0 });

            document.querySelectorAll('.station-link').forEach(link => observer.observe(link));

            const scrap = document.getElementById('scrap-indicator');

                    scrap.style.opacity = '1';

                    scrap.classList.add('animate-scrap-wiggle');

                }

            });

        };
